                                          Usage

                                          WARNING:The plugins section is still under development and its capabilities may change rapidly. So until the development of plugins is completed, expect a sudden change in its features.

                                          Plugins Prop accepts an array of plugins, which the following two properties can be entered on each of them.

                                          Propdefault
                                          position"right"
                                          disabledfalse

                                          This is the general way to add a imported plugin to Datepicker:

                                          import React from "react"
import DatePicker from "react-multi-date-picker"
import { MyFavoritePlugin } from "react-mylti-date-picker/plugins"

export default function Example() {
  return (
    <DatePicker
      plugins={[
        <MyFavoritePlugin />
      ]}
    />
  )
}

                                          Date Click Listener
                                          import { DatePanel } from "react-multi-date-picker/plugins"
const [value, setValue] = useState([])
.
.
.
<DatePicker
  value={value}
  onChange={setValue}
  plugins={[
    <DatePanel
      removeButton={false}
      onDateClicked={dateObject => {
        let mustDelete = window.confirm("are you sure you want to delete this item ?")

        if (mustDelete) setValue(
          value.filter(date => date !== dateObject)
        )
      }}
    />
  ]}
/>
